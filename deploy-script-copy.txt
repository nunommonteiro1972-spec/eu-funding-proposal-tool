async function deployFunction(name) {
    console.log(`Deploying function: ${name}...`);
    const functionDir = join(process.cwd(), 'supabase', 'functions', name);
    const indexPath = name === 'server' ? join(functionDir, 'bundle.ts') : join(functionDir, 'index.ts');

    try {
        const content = readFileSync(indexPath, 'utf8');

        // Self-hosted Supabase usually doesn't have the Management API for functions
        // exposed in the same way as cloud. 
        // We might need to use the Docker API or just copy the files.
        // However, some setups (like Elestio) expose the functions API.

        const response = await fetch(`${SUPABASE_URL}/functions/v1/${name}`, {
            method: 'POST',
            headers: {
                'Authorization': `Bearer ${SERVICE_ROLE_KEY}`,
                'Content-Type': 'application/typescript',
                'x-upsert': 'true'
            },
            body: content
        });

        const responseText = await response.text();
        console.log(`Response for ${name}: ${response.status} ${response.statusText}`);
        console.log(`Body: ${responseText}`);
    } catch (error) {
        console.error(`‚ùå Failed to deploy ${name}:`, error.message);
    }
}

const functions = ['server', 'proposal-copilot', 'search-funding'];
for (const f of functions) {
    await deployFunction(f);
}
